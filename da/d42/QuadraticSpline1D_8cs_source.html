56  public QuadraticSpline1D(SortedList<float, float> points)
57  {
58  if (points.Count < 2)
59  {
60  if (points.Count == 1)
61  {
62  throw new ArgumentException("points","List contains only a single point. A spline must have at least two points.");
63  }
64  else
65  {
66  throw new ArgumentException("points","List is empty. A spline must have at least two points.");
67  }
68  }
69 
70  Points = new SortedPointsList<float>(points);
71 
72  // Calculate the coefficients of the spline:
73  _parameters = new float[points.Count];
74 
75  // Find the first segment parameter, which will be a straight line:
76  {
77  float dx = Points.Key[1] - Points.Key[0];
78  float dy = Points.Value[1] - Points.Value[0];
79 
80  _parameters[0] = dy/dx;
81  }
82 
83  // Recursively find the other _parameters:
84  for (int i = 1; i < points.Count; i++)
85  {
86  float dx = Points.Key[i] - Points.Key[i-1];
87  float dy = Points.Value[i] - Points.Value[i-1];
88 
89  _parameters[i] = -_parameters[i-1] + 2.0f*dy/dx;
90  }

110  public float GetNthDerivativeAt(float x, uint derivative)
111  {
112  // The input parameter must lie between the outer points, and must not be NaN:
113  if (!( x >= Points.Key[0] && x <= Points.Key[Points.Count - 1]))
114  {
115  throw new ArgumentOutOfRangeException("x","Cannot interpolate outside the interval given by the spline points.");
116  }
117 
118  // Find the index `i` of the closest point to the right of the input `x` parameter, which is the right point
119  // used to interpolate between. Therefore, `i-1` indicates the left point of the interval.
120  int i = Points.Key.BinarySearch(x);
121 
122  // BinarySearch returns a bitwise complement of the index if the point is not exactly in the list, such as
123  // when interpolating. To turn it into a valid index, we take the bitwise complement again if it is negative:
124  if (i < 0)
125  {
126  i = ~i;
127  }
128 
129  // If the index is zero, we are exactly on the first point in the list. We increment by one to get the value
130  // of the first spline segment, to avoid an IndexOutOfRangeException later on:
131  if (i == 0)
132  {
133  i++;
134  }
135 
136  float x1 = Points.Key[i-1];
137  float x2 = Points.Key[i];
138  float y1 = Points.Value[i-1];
139  float y2 = Points.Value[i];
140 
141  // Calculate and return the interpolated value:
142  float dx = x2 - x1;
143  float dy = y2 - y1;
144 
145  float a = -_parameters[i]*dx + dy;
146  float t = (x-x1)/dx;
147 
148  // Return a different function depending on the derivative level:
149  switch (derivative)
150  {
151  case 0: return (1.0f - t)*y1 + t*y2 + t*(1.0f-t)*a;
152  case 1: return (dy + a * (1.0f - 2.0f * t))/dx;
153  case 2: return (2.0f * a * t)/(dx*dx);
154  default: return 0.0f;
155  }
156  }

187  public override IEnumerable<float> SolveRaytrace(QuarticFunction surfaceFunction, float z0 = 0.0f, float c = 1.0f)
188  {
189  // Solve the polynomial equation for each segment:
190  for (int i = 1; i < Points.Count; i++)
191  {
192  Real x1 = Points.Key[i-1];
193  Real x2 = Points.Key[i];
194  Real y1 = Points.Value[i-1];
195  Real y2 = Points.Value[i];
196 
197  // Calculate and return the interpolated value:
198  Real dx = x2 - x1;
199  Real div = 1.0/dx;
200  Real dy = y2 - y1;
201 
202  Real a = -_parameters[i]*dx + dy;
203 
204  // Write in the form of a0 + a1 z + a2 z^2:
205  Real a0 = -a*x1*x1*div*div - (a + dy)*x1*div + y1;
206  Real a1 = 2.0*a*x1*div*div + (a + dy)*div;
207  Real a2 = -a*div*div;
208 
209  // Substitute z = z0 + c t:
210  Real A0 = a0 + a1*z0 + a2*z0*z0;
211  Real A1 = (a1 + 2.0*a2*z0)*c;
212  Real A2 = a2*c*c;
213 
214  // Find the quartic polynomial to solve:
215  Real p0 = surfaceFunction.a0 - A0*A0;
216  Real p1 = surfaceFunction.a1 - 2.0*A0*A1;
217  Real p2 = surfaceFunction.a2 - (2.0*A0*A2 + A1*A1);
218  Real p3 = surfaceFunction.a3 - 2.0*A1*A2;
219  Real p4 = surfaceFunction.a4 - A2*A2;
220 
221  // Solve the quartic polynomial:
222  IEnumerable<float> intersections = QuarticFunction.Solve((float)p0, (float)p1, (float)p2, (float)p3, (float)p4);
223 
224  // Only return the value if it is sampled within the segment that we are currently considering,
225  // otherwise the value we got is invalid:
226  foreach (var j in intersections)
227  {
228  if (((z0 + c*j) > x1) && ((z0 + c*j) <= x2))
229  {
230  yield return j;
231  }
232  }
233  }
234  }
