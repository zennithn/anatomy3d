53  [Fact]
54  public void TestEqualsFunction()
55  {
56  Assert.True(new Real(0).Equals(new Real(0)));
57  Assert.False(new Real(100).Equals(new Real(150)));
58  Assert.True(new Real(500).Equals(new Real(500.0)));
59  Assert.True(Real.PositiveInfinity.Equals(Real.PositiveInfinity));
60  Assert.True(Real.NegativeInfinity.Equals(Real.NegativeInfinity));
61  Assert.False(Real.PositiveInfinity.Equals(Real.NegativeInfinity));
62  // Although NaN compares as unequal to itself when using the == operator, the .Equals method <i>does</i>
63  // detect NaNs as equal to themselves. Anything else would break the IEquatable/IEquatable<T> contract
64  // and also break using this as a hash table key.
65  Assert.True(Real.NaN.Equals(Real.NaN));
66  }

68  [Fact]
69  public void TestEqualsOperator()
70  {
71  Assert.True(new Real(0) == new Real(0));
72  Assert.False(new Real(100) == new Real(150));
73  Assert.True(new Real(500) == new Real(500.0));
74  Assert.True(Real.PositiveInfinity == Real.PositiveInfinity);
75  Assert.True(Real.NegativeInfinity == Real.NegativeInfinity);
76  Assert.False(Real.PositiveInfinity == Real.NegativeInfinity);
77 
78  // When using the == operator, NaN is unequal to itself, as specified in the IEEE 754 standard for
79  // floating-point arithmetic.
80  Assert.False(Real.NaN == Real.NaN);
81  }
