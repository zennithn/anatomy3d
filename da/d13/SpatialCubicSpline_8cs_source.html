67  public SpatialCubicSpline(SortedList<float, Vector3> points)
68  {

126 #region Private Methods
133  private void GenerateCurveNormal()
134  {
135  // Generate a tangent at key points:
136  // TODO: make the precision adapt to the precision of centerCurve!
137 
138  SortedList<float, float> normalX = new SortedList<float, float>();
139  SortedList<float, float> normalY = new SortedList<float, float>();
140  SortedList<float, float> normalZ = new SortedList<float, float>();
141 
142  // No normal supplied, pick arbitrary normal vector and calculate a binormal:
143  Vector3 binormal = Vector3.Cross(VectorUtilities.InventNormal(GetTangentAt(0.0f)), GetTangentAt(0.0f));
144 
145  // Traverse through all the points in the spline, and calculate the next normal, maintaining smoothness by
146  // memorizing the previous normal:
147 
148  float previousValue = X.Points.Key[0]; // X.Points.Key is guaranteed to have at least 2 entries, this is safe.
149  for (int i = 1; i < X.Points.Count; i++)
150  {
151  float currentValue = X.Points.Key[i];
152  float segmentSize = currentValue - previousValue;
153 
154  // Interpolate between each two points:
155  for (int j = 0; j < kInterpolationStepCount; j++)
156  {
157  float t = previousValue + segmentSize*(float)j/(float)kInterpolationStepCount;
158 
159  Vector3 direction = Vector3.Normalize(GetTangentAt(t));
160  Vector3 normal = Vector3.Normalize(Vector3.Cross(direction, binormal));
161  binormal = Vector3.Cross(normal, direction);
162 
163  normalX.Add(t, normal.X);
164  normalY.Add(t, normal.Y);
165  normalZ.Add(t, normal.Z);
166  }
167 
168  previousValue = currentValue;
169  }
170 
171  // Add the final point as well:
172  {
173  float t = X.Points.Key[X.Points.Count - 1];
174 
175  Vector3 direction = Vector3.Normalize(GetTangentAt(t));
176  Vector3 normal = Vector3.Normalize(Vector3.Cross(direction, binormal));
177  binormal = Vector3.Cross(normal, direction);
178 
179  normalX.Add(t, normal.X);
180  normalY.Add(t, normal.Y);
181  normalZ.Add(t, normal.Z);
182  }
183 
184  this.NormalX = new CubicSpline1D(normalX);
185  this.NormalY = new CubicSpline1D(normalY);
186  this.NormalZ = new CubicSpline1D(normalZ);
187  }
188 #endregion Private Methods
