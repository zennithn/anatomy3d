171  public override List<Vertex> GenerateVertexList(int resolutionU, int resolutionV)
172  {
173  List<Vertex> output = new List<Vertex>(CalculateVertexCount(resolutionU, resolutionV));
174 
175  // Load all required variables:
176  Vector3 up = new Vector3(0.0f, 0.0f, 1.0f);
177 
178  Vector3 pointTangent = Direction;
179  Vector3 pointNormal = Normal;
180  Vector3 pointBinormal = Binormal;
181 
182  Vector3 translation = Center;
183 
184  // Get the radius at the top of the hemisphere:
185  float topRadius = this.Radius.GetValueAt(new Vector2(0.0f, 0.5f*(float)Math.PI));
186 
187  // Generate the first point at the pole of the hemisphere:
188  output.Add(new Vertex(translation + topRadius*pointTangent, pointTangent));
189 
190  // Generate rings of the other points:
191  for (int j = 1; j < (resolutionV+1); j++)
192  {
193  for (int i = 0; i < resolutionU; i++)
194  {
195  // First find the normalized uv-coordinates, u = [0, 2pi], v = [0, 1/2pi]:
196  float u = (float)i/(float)resolutionU * 2.0f * (float)Math.PI;
197  float v = (float)j/(float)resolutionV * 0.5f * (float)Math.PI;
198 
199  // Calculate the position of the rings of vertices:
200  float x = (float)Math.Sin(v) * (float)Math.Cos(u);
201  float y = (float)Math.Sin(v) * (float)Math.Sin(u);
202  float z = (float)Math.Cos(v);
203 
204  float radius = this.Radius.GetValueAt(new Vector2(u, v));
205 
206  Vector3 surfaceNormal = x*pointNormal + y*pointBinormal + z*pointTangent;
207  Vector3 surfacePosition = translation + radius*surfaceNormal;
208 
209  output.Add(new Vertex(surfacePosition, surfaceNormal));
210  }
211  }
212 
213  // Recalculate the surface normal after deformation:
214  for (int j = 1; j < resolutionV; j++)
215  {
216  for (int i = 0; i < (resolutionU - 1); i++)
217  {
218  Vector3 surfacePosition = output[(j-1)*resolutionU + i + 1].Position;
219  Vector3 du = surfacePosition - output[(j-1)*resolutionU + i + 1 + 1].Position;
220  Vector3 dv = surfacePosition - output[(j)*resolutionU + i + 1].Position;
221 
222  // Calculate the position of the rings of vertices:
223  Vector3 surfaceNormal = Vector3.Cross(Vector3.Normalize(du), Vector3.Normalize(dv));
224 
225  output[(j-1)*resolutionU + i + 1] = new Vertex(surfacePosition, surfaceNormal);
226  }
227 
228  // Stitch the end of the triangles:
229  Vector3 surfacePosition2 = output[(j-1)*resolutionU + resolutionU].Position;
230  Vector3 du2 = surfacePosition2 - output[(j-1)*resolutionU + 1].Position;
231  Vector3 dv2 = surfacePosition2 - output[(j)*resolutionU + resolutionU].Position;
232 
233  // Calculate the position of the rings of vertices:
234  Vector3 surfaceNormal2 = Vector3.Cross(Vector3.Normalize(du2), Vector3.Normalize(dv2));
235 
236  output[(j-1)*resolutionU + resolutionU] = new Vertex(surfacePosition2, surfaceNormal2);
237  }
238 
239  return output;
240  }
