210  public override List<Vertex> GenerateVertexList(int resolutionU, int resolutionV)
211  {
212  var roughs = ParallelEnumerable.Range(0, resolutionV + 1).AsOrdered().SelectMany((j =>
213  {
214  float v = (float) j / (float) resolutionV;
215 
216  // Find the values at each ring:
217  Vector3 curveTangent = Vector3.Normalize(CenterCurve.GetTangentAt(v));
218  Vector3 curveNormal = Vector3.Normalize(CenterCurve.GetNormalAt(v));
219  Vector3 curveBinormal = Vector3.Cross(curveTangent, curveNormal);
220 
221  Vector3 translation = CenterCurve.GetPositionAt(v);
222 
223  float startAngle = StartAngle.GetValueAt(v);
224  float endAngle = EndAngle.GetValueAt(v);
225 
226  return Enumerable.Range(0, resolutionU).Select((i) =>
227  {
228  float u = startAngle + (endAngle - startAngle) * (float) i / (float) resolutionU;
229 
230  float radius = Radius.GetValueAt(new Vector2(u, v));
231 
232  // Calculate the position of the rings of vertices:
233  Vector3 surfaceNormal = (float) Math.Cos(u) * curveNormal + (float) Math.Sin(u) * curveBinormal;
234  Vector3 surfacePosition = translation + radius * surfaceNormal;
235  return new Vertex(surfacePosition, surfaceNormal);
236  });
237  }));
238 
239  List<Vertex> output = roughs.ToList();
240 
241  // Recalculate the surface normal after deformation:
242  for (int j = 1; j < resolutionV; j++)
243  {
244  for (int i = 0; i < (resolutionU - 1); i++)
245  {
246  Vector3 surfacePosition = output[(j-1)*resolutionU + i].Position;
247  Vector3 du = surfacePosition - output[(j-1)*resolutionU + i + 1].Position;
248  Vector3 dv = surfacePosition - output[(j)*resolutionU + i].Position;
249 
250  // Calculate the position of the rings of vertices:
251  Vector3 surfaceNormal = Vector3.Cross(Vector3.Normalize(du), Vector3.Normalize(dv));
252 
253  output[(j-1)*resolutionU + i] = new Vertex(surfacePosition, surfaceNormal);
254  }
255 
256  // Stitch the end of the triangles:
257  Vector3 surfacePosition2 = output[(j-1)*resolutionU + resolutionU-1].Position;
258  Vector3 du2 = surfacePosition2 - output[(j-1)*resolutionU].Position;
259  Vector3 dv2 = surfacePosition2 - output[(j)*resolutionU + resolutionU-1].Position;
260 
261  // Calculate the position of the rings of vertices:
262  Vector3 surfaceNormal2 = Vector3.Cross(Vector3.Normalize(du2), Vector3.Normalize(dv2));
263 
264  output[(j-1)*resolutionU + resolutionU-1] = new Vertex(surfacePosition2, surfaceNormal2);
265  }
266 
267  return output;
268  }
