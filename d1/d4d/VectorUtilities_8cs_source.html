32  public static Vector3 InventNormal(Vector3 v)
33  {
34  // If the vector has infinite or NaN components, throw an error:
35  if (Single.IsNaN(Vector3.Dot(Vector3.One, v)) || Single.IsInfinity(Vector3.Dot(Vector3.One, v)))
36  {
37  throw new ArgumentException("Vector must not contain infinite or NaN values.", "v");
38  }
39 
40  // TODO: Here it would be nice to add IsNormal instead of an arbitrary comparison, but alas... it's only
41  // supported in .NET 5.0 which we can't target yet.
42  if ((Math.Abs(v.X) < 0.001) && (Math.Abs(v.Y) < 0.001) && (Math.Abs(v.Z) < 0.001))
43  {
44  throw new ArgumentException("Vector is too small.", "v");
45  }
46 
47  Vector3 normalizedV = Vector3.Normalize(v);
48  Vector3 up = Vector3.UnitZ;
49 
50  // If the vector is pointing in the same direction as the up vector, the cross product can not be trusted.
51  // So we use a different up vector.
52  if (Math.Sign(normalizedV.Z) > 1/Math.Sqrt(2))
53  {
54  up = Vector3.UnitY;
55  }
56 
57  return Vector3.Normalize(Vector3.Cross(normalizedV, up));
58  }
59  }
