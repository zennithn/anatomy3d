57  {
58  if (points.Count < 2)
59  {
60  if (points.Count == 1)
61  {
62  throw new ArgumentException("List contains only a single point. A spline must have at least two points.", "points");
63  }
64  else
65  {
66  throw new ArgumentException("List is empty. A spline must have at least two points.", "points");
67  }
68  }
69 
70  Points = new SortedPointsList<float>(points);
71 
72  // Calculate the coefficients of the spline:
73  float[] a = new float[points.Count];
74  float[] b = new float[points.Count];
75  float[] c = new float[points.Count];
76  float[] d = new float[points.Count];
77 
78  // Set up the boundary condition for a natural spline:
79  {
80  float x2 = 1.0f/(Points.Key[1] - Points.Key[0]);
81  float y2 = Points.Value[1] - Points.Value[0];
82 
83  a[0] = 0.0f;
84  b[0] = 2.0f*x2;
85  c[0] = x2;
86  d[0] = 3.0f*(y2*x2*x2);
87  }
88 
89  // Set up the tridiagonal matrix linear system:
90  for (int i = 1; i < points.Count-1; i++)
91  {
92  float x1 = 1.0f/(Points.Key[i] - Points.Key[i-1]);
93  float x2 = 1.0f/(Points.Key[i+1] - Points.Key[i]);
94 
95  float y1 = Points.Value[i] - Points.Value[i-1];
96  float y2 = Points.Value[i+1] - Points.Value[i];
97 
98  a[i] = x1;
99  b[i] = 2.0f*(x1 + x2);
100  c[i] = x2;
101  d[i] = 3.0f*(y1*x1*x1 + y2*x2*x2);
102  }
103 
104  // Set up the boundary condition for a natural spline:
105  {
106  float x1 = 1.0f/(Points.Key[points.Count-1] - Points.Key[points.Count-2]);
107  float y1 = (Points.Value[points.Count-1] - Points.Value[points.Count-2]);
108

135  public float GetNthDerivativeAt(float x, uint derivative)
136  {
137  // The input parameter must lie between the outer points, and must not be NaN:
138  if (!( x >= Points.Key[0] && x <= Points.Key[Points.Count - 1]))
139  {
140  throw new ArgumentOutOfRangeException("x","Cannot interpolate outside the interval given by the spline points.");
141  }
142 
143  // Find the index `i` of the closest point to the right of the input `x` parameter, which is the right point
144  // used to interpolate between. Therefore, `i-1` indicates the left point of the interval.
145  int i = Points.Key.BinarySearch(x);
146 
147  // BinarySearch returns a bitwise complement of the index if the point is not exactly in the list, such as
148  // when interpolating. To turn it into a valid index, we take the bitwise complement again if it is negative:
149  if (i < 0)
150  {
151  i = ~i;
152  }
153 
154  // If the index is zero, we are exactly on the first point in the list. We increment by one to get the value
155  // of the first spline segment, to avoid an IndexOutOfRangeException later on:
156  if (i == 0)
157  {
158  i++;
159  }
160 
161  float x1 = Points.Key[i-1];
162  float x2 = Points.Key[i];
163  float y1 = Points.Value[i-1];
164  float y2 = Points.Value[i];
165 
166  // Calculate and return the interpolated value:
167  float dx = x2 - x1;
168  float dy = y2 - y1;
169 
170  float a = parameters[i-1]*dx - dy;
171  float b = -parameters[i]*dx + dy;
172  float t = (x-x1)/dx;
173 
174  // Return a different function depending on the derivative level:
175  switch (derivative)
176  {
177  case 0: return (1.0f - t)*y1 + t*y2 + t*(1.0f-t)*((1.0f-t)*a + t*b);
178  case 1: return (y2 - y1 + 3.0f*(a-b)*t*t + (2.0f*b - 4.0f*a)*t + a)/dx;
179  case 2: return (a*(6.0f*t - 4.0f) + b*(1.0f-3.0f*t))/(dx*dx);
180  case 3: return 6.0f*(a-b)/(dx*dx*dx);
181  default: return 0.0f;
182  }
183  }

212  private static float[] ThomasAlgorithm(float[] a, float[] b, float[] c, float[] d)
213  {
214  int size = d.Count();
215 
216  // Perform forward sweep:
217  float[] newC = new float[size];
218  float[] newD = new float[size];
219 
220  newC[0] = c[0] / b[0];
221  newD[0] = d[0] / b[0];
222  for (int i = 1; i < size; i++)
223  {
224  newC[i] = c[i] / ( b[i] - a[i]*newC[i-1] );
225  newD[i] = (d[i] - a[i]*newD[i-1]) / ( b[i] - a[i]*newC[i-1] );
226  }
227 
228  // Perform back substitution:
229  float[] x = new float[size];
230 
231  x[size-1] = newD[size-1];
232  for (int i = (size - 2); i >= 0; i--)
233  {
234  x[i] = newD[i] - newC[i]*x[i+1];
235  }
236 
237  return x;
238  }
