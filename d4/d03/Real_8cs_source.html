436  public bool IsNormal
437  {
438 #if REALTYPE_DECIMAL
439  get => _v != new Decimal(0);
440 #elif REALTYPE_FLOAT || REALTYPE_DOUBLE
441  get
442  {
443  #if REALTYPE_FLOAT
444  int expBits = kExponentMask & BitConverter.ToInt32(BitConverter.GetBytes(_v), 0);
445  #else // double
446  long expBits = kExponentMask & BitConverter.DoubleToInt64Bits(_v);
447  #endif
448  // Normal numbers are:
449  // * nonzero
450  // * not denormalized
451  // * noninfinite
452  // * numbers
453  // Maximum exponent represents infinity or NaN, minimum exponent is 0 or denormalized.
454  return (expBits != kExponentMask) && (expBits != 0);
455  }
456 #else

458 #endif

504  private bool? NearSpecialCases(Real o)
505  {
506  if (IsNaN || o.IsNaN)
507  {
508  // NaN is not similar to anything else - even itself.
509  return false;
510  }
511 
512  if (IsInfinity)
513  {
514  // Infinities are only near the exactly identical infinity,
515  // regardless of the slop factor.
516  return Equals(o);
517  }
518 
519  if (o.IsInfinity)
520  {
521  // this isn't infinity or we would have gotten into the previous case.
522  return false;
523  }
524 
525  return null;
526  }

535  public bool IsAbsolutelyNear(Real o, Real slop)
536  {
537  if (slop < 0.0)
538  {
539  throw new ArgumentException("slop cannot be negative.","slop");
540  }
541  bool? sc = NearSpecialCases(o);
542  if (sc != null)
543  {
544  return sc.Value;
545  }
546 
547  Real diff = this - o;
548  if (diff < 0.0) diff = -diff;
549  return diff <= slop;
550  }
551 
561  public bool IsRelativelyNear(Real o, Real slop)
562  {
563  if (slop < 0.0)
564  {
565  throw new ArgumentException("slop cannot be negative.", "slop");
566  }
567  bool? sc = NearSpecialCases(o);
568  if (sc != null)
569  {
570  return sc.Value;
571  }
572 
573  if (o == 0.0)
574  {
575  // Zero is not relatively near anything else.
576  return this == 0.0;
577  }
578 
579  Real big, little;
580  if (this < 0.0)
581  {
582  if (o > 0.0) return false;
583  // Negative values; "big" and "little" refer to magnitude and the signs will cancel
584  // out during division, so reverse the obvious semantics for which goes where.
585  if (this > o)
586  {
587  big = o;
588  little = this;
589  }
590  else
591  {
592  big = this;
593  little = o;
594  }
595  }
596  else
597  {
598  if (o < 0.0) return false;
599  if (this > o)
600  {
601  big = this;
602  little = o;
603  }
604  else
605  {
606  big = o;
607  little = this;
608  }
609  }
610 
611  Real ratio = big / little;
612  return (ratio - 1.0) <= slop;
613  }
