78  public override void _Input(InputEvent @event)
79  {
80  if (@event is InputEventMouseButton eventMouseButton)
81  {
82  // If there is a scrollwheel event, zoom the camera in or out:
83  if (eventMouseButton.IsPressed()){
84  if (eventMouseButton.ButtonIndex == (int)ButtonList.WheelUp){
85  // Divide by the zoom factor to zoom in:
86  distanceFromFocus /= zoomFactor;
87  }
88  if (eventMouseButton.ButtonIndex == (int)ButtonList.WheelDown){
89  // Multiply by the zoom factor to zoom out:
90  distanceFromFocus *= zoomFactor;
91  }
92  }
93  }
94  else if (@event is InputEventMouseMotion eventMouseMotion)
95  {
96  if (Input.IsMouseButtonPressed(3) && Input.IsKeyPressed((int)KeyList.Shift))
97  {
98  // If the user moves the cursor while holding the scrollwheel button and holding
99  // the Shift key, move the focus (target) position:
100  Vector3 screenNormal = (Translation - focusPosition).Normalized();
101  Vector3 screenRight = screenNormal.Cross(Vector3.Up).Normalized();
102  Vector3 screenUp = -screenNormal.Cross(screenRight).Normalized();
103  float speed = movementSpeed*distanceFromFocus;
104  focusPosition += speed*eventMouseMotion.Relative.x*screenRight;
105  focusPosition += speed*eventMouseMotion.Relative.y*screenUp;
106  } else if (Input.IsMouseButtonPressed(3))
107  {
108  // If the user moves the cursor while holding the scrollwheel button without
109  // modifiers, rotate the camera around the target:
110  angularPosition.x -= horizontalSpeed*eventMouseMotion.Relative.x;
111  angularPosition.y -= verticalSpeed*eventMouseMotion.Relative.y;
112  }
113  }
114 
115  // Implement Gimbal lock:
116  double u = angularPosition.x;
117  double v = Mathf.Max(Mathf.Min(0.999f*Mathf.Pi, angularPosition.y), 0.001f);
118 
119  // Calculate the 3D camera position from the angular position:
120  Vector3 position = distanceFromFocus*(new Vector3((float)(Math.Sin(v)*Math.Sin(u)), (float)Math.Cos(v), (float)(Math.Sin(v)*Math.Cos(u))));
121 
122  // Set the camera's position and rotation:
123  LookAtFromPosition(focusPosition + position, focusPosition, Vector3.Up );
124  }
