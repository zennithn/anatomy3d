143  [InlineData(0f, 1f, 1f, new float[]{-1f, 0f})]
144  public void TestSolve(float a0, float a1, float a2, float[] want)
145  {
146  IList<float> got = QuadraticFunction.Solve(a0, a1, a2).ToList();
147  Assert.Equal(want.Length, got.Count);
148  // There are no guarantees as to the order of values returned, so sort both sequences and compare items
149  // at equal locations in the sorted sequence. Expect them all to match. We don't need to try any other
150  // permutations of these lists to find out if there's some way they do match:
151  //
152  // If the smallest unmatched elements of each set are not adequately near each other, there is no way to
153  // match the smaller of the two to any element in its other set. We have just compared it against the
154  // smallest element in the other set, and found it to be too large. It cannot be too small, because we
155  // are specifically considering the smaller element of the mismatched pair. Since the too-large element
156  // was a smallest element of its set, all other elements are at least that large. Therefore, there is no
157  // element small enough to match the smallest unmatched element between both sets, so the sets are unequal.
158  // Backtracking does not help, because to find a smaller element of the opposite set, we must take it away
159  // from its match to an element that is no larger than the one we are now matching (because all larger
160  // elements are still in the unmatched portion of the set), and now _that_ element has the same problem
161  // but worse.
162  //
163  // Therefore, there is no reason to attempt any other order than an element-by-element pairwise comparison
164  // of the sorted sequences. ∎
165  Array.Sort(want);
166  int i = 0;
167  var gotSorted = from c in got orderby c select c;
168  foreach(float f in gotSorted)
169  {
170  Assert.True(Near(want[i++], f));
171  }
172  }
173  }
